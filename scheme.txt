## テーブル
- 速度・出力管理テーブル
CREATE TABLE speed_output_curve(
    curve_id INTEGER PRIMARY KEY AUTOINCREMENT,
    curve_group_id INTEGER NOT NULL,
    speed INTEGER NOT NULL CHECK(speed >= 0),
    output INTEGER NOT NULL CHECK(output >= 0)
);
speedの最大値が最高速度

- 速度・加速度管理テーブル
CREATE TABLE speed_accel_curve(
    curve_id INTEGER PRIMARY KEY AUTOINCREMENT,
    accel REAL NOT NULL CHECK(accel >= 0),
    speed INTEGER NOT NULL CHECK(speed >= 0)
);
